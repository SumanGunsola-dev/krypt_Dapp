   const sendTransaction = async () => {
        try {
            if (!ethereum) return alert("Please install metamask");
            const { addressTo, amount, keyword, message } = formData;
            const { provider, signer, transactionContract } = getEthereumContract();
            const parsedAmount = ethers.utils.parseEther(amount);

            const tx = await signer.sendTransaction({
                to: addressTo,
                value: parsedAmount,
                // Optional: gasLimit if needed (ethers estimates automatically)
            });
            const transactionHash = await transactionContract.addToBlockchian(addressTo, parsedAmount, message, keyword);
            // console.log("Transaction hash:", tx.hash);

            // console.log("Transaction confirmed!");
            setIsLoading(true);
            console.log(`Loading - ${transactionHash.hash}`);
            await transactionHash.wait(); // wait until it's mined
            setIsLoading(false);
            console.log(`Success - ${transactionHash.hash}`);

            const transactionCount = await transactionContract.getTransactionCount();
            setTransactionCount(transactionCount.toNumber())
        } catch (error) {
            console.log("Transaction failed:", error);
            throw new Error("No ethereum object. ");
        }
    }
